188  template<typename Event, typename Track, typename Pair>
189  std::vector<Pair> JJFemtoMixer<Event,Track,Pair>::MakePairs(const std::vector<Track> &tracks)
190  {
191  bool reverse = false;
192  const std::size_t trackVecSize = tracks.size();
193 
194  std::vector<Pair> tmpVector;
195  tmpVector.reserve((1 + trackVecSize) * trackVecSize / 2); // reserve the expected vector size (arithmetic sum, from 1 to N)
196 
197  for (std::size_t iter1 = 0; iter1 < trackVecSize; ++iter1)
198  for (std::size_t iter2 = iter1 + 1; iter2 < trackVecSize; ++iter2)
199  {
200  if (reverse)
201  tmpVector.push_back(Pair(tracks.at(iter2),tracks.at(iter1)));
202  else
203  tmpVector.push_back(Pair(tracks.at(iter1),tracks.at(iter2)));
204 
205  reverse = !reverse; // reverse the order of tracks every other time (get rid of the bias from the track sorter)
206  }
207 
208  return tmpVector;
209  }

212  std::map<std::size_t, std::vector<Pair> > JJFemtoMixer<Event,Track,Pair>::SortPairs(const std::vector<Pair> &pairs)
213  {
214  std::size_t currentPairHash = 0;
215  std::map<std::size_t, std::vector<Pair> > pairMap;
216 
217  for (const auto &pair : pairs)
218  {
219  if (fPairCutFunction(pair))
220  continue;
221 
222  currentPairHash = fPairHashingFunction(pair);
223 
224  // an entry for given currentPairHash may already exist so we must check if that's the case
225  if (pairMap.find(currentPairHash) != pairMap.end())
226  {
227  pairMap.at(currentPairHash).push_back(pair);
228  }
229  else
230  {
231  pairMap.emplace(currentPairHash,std::vector<Pair>(1,pair));
232  }
233  }
234 
235  return pairMap;

250  std::map<std::size_t, std::vector<Pair> > JJFemtoMixer<Event,Track,Pair>::AddEvent(const Event &event, const std::vector<Track> &tracks)
251  {
252  std::uniform_int_distribution<int> dist(0,tracks.size()-1);
253  std::size_t evtHash = fEventHashingFunction(event);
254  std::pair<Event, Track> trackPair{event,tracks.at(dist(fGenerator))};
255 
256  // an entry for given evtHash may not exist, so we must check if that's the case
257  if (fSimilarityMap.find(evtHash) == fSimilarityMap.end())
258  {
259  fSimilarityMap.emplace(evtHash,std::deque<std::pair<Event, Track> >(1,trackPair));
260  }
261  else
262  {
263  fSimilarityMap.at(evtHash).push_back(trackPair);
264  if (fSimilarityMap.at(evtHash).size() > fBufferSize)
265  fSimilarityMap.at(evtHash).pop_front();
266  }
267 
268  return SortPairs(MakePairs(tracks));
269  }

282  }
