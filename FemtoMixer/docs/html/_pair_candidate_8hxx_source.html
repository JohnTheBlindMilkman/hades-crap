186  void CFKinematics(const TrackCandidate &part1, const TrackCandidate &part2)
187  {
188  // adapted from https://github.com/DanielWielanek/HAL/blob/main/analysis/femto/base/FemtoPairKinematics.cxx
189  Double_t tPx = part1.Px + part2.Px;
190  Double_t tPy = part1.Py + part2.Py;
191  Double_t tPz = part1.Pz + part2.Pz;
192  Double_t tE = part1.Energy + part2.Energy;
193  Double_t tPt = tPx * tPx + tPy * tPy;
194  Double_t tMt = tE * tE - tPz * tPz; // mCVK;
195  tMt = TMath::Sqrt(tMt);
196  Kt = TMath::Sqrt(tPt);
197  Double_t tBeta = tPz / tE;
198  Double_t tGamma = tE / tMt;
199 
200  // Transform to LCMS
201 
202  Double_t particle1lcms_pz = tGamma * (part1.Pz - tBeta * part1.Energy);
203  Double_t particle1lcms_e = tGamma * (part1.Energy - tBeta * part1.Pz);
204  Double_t particle2lcms_pz = tGamma * (part2.Pz - tBeta * part2.Energy);
205  Double_t particle2lcms_e = tGamma * (part2.Energy - tBeta * part2.Pz);
206 
207  // Rotate in transverse plane
208 
209  Double_t particle1lcms_px = (part1.Px * tPx + part1.Py * tPy) / Kt;
210  Double_t particle1lcms_py = (-part1.Px * tPy + part1.Py * tPx) / Kt;
211 
212  Double_t particle2lcms_px = (part2.Px * tPx + part2.Py * tPy) / Kt;
213  Double_t particle2lcms_py = (-part2.Px * tPy + part2.Py * tPx) / Kt;
214 
215  QOut = abs(particle1lcms_px - particle2lcms_px);
216  QSide = abs(particle1lcms_py - particle2lcms_py);
217  QLong = abs(particle1lcms_pz - particle2lcms_pz);
218  Double_t mDE = particle1lcms_e - particle2lcms_e;
219  QInv = TMath::Sqrt(TMath::Abs(QOut * QOut + QSide * QSide + QLong * QLong - mDE * mDE));
220  }
